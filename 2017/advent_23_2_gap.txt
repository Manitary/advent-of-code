b:=99*100+100000;;c:=b+17000;;h:=0;;
Print("b: ",b,"; c: ",c,"\n");
for n in [b, b+17..c] do
	check:=false;;
	Print("n: ",n,"\n");
	for d in [2..b] do
		if 2*d > n then
			break;
		fi;
		for e in [2..b] do
			if d*e <= n then
				if d*e = n then
					h:=h+1;;
					Print("d: ",d,"; e: ",e,"; h: ",h,"\n");
					check:=true;;
					break;
				fi;
			fi;
		od;
		if check then
			break;
		fi;
	od;
od;
